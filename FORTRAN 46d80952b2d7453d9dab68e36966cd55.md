### Seno

$$
\sin(x) = x - \frac{x^3}{3!}+  \frac{x^5}{5!} - \cdots
= \sum_{n=0}^{\infty} \frac{(-1)^n}{(2n+1)!} x^{2n+1}
$$

### Cosseno

$$
\cos(x) = 1 - \frac{x^2}{2!} + \frac{x^4}{4!} + \dots = \sum_{n=0}^\infty \frac{(-1)^n}{2n!}x^{2n}
$$

Calcular as ra√≠zes de uma equa√ß√£o do segundo grau

```fortran
subroutine quadratica(a,b,c,nr,r1,r2)
!----------------------------------------------------
!         Resolve a equa√ß√£o d segundo grau:         !
!                    ax^2+bx+c=0                    !
!r1: raiz; r2: raiz; nr: numero de raizes da equa√ß√£o!
!----------------------------------------------------
    implicit none
    real, intent(in) :: a,b,c
    real, intent(out) ::  r1,r2
    integer, intent(out) :: nr
    real :: delta !argumento dentro da raiz quadrada

    delta = b**2 - 4*a*c

    if (delta<0.) then !se delta for negativo, n√£oo h√° solu√ß√£o nr=0
        nr = 0
    else if (delta >0.) then !delta positivo retorna duas ra√≠zes
        nr = 2
        r1 = (-b - sqrt(delta))/(2*a)
        r2 = (-b + sqrt(delta))/(2*a)
    else
        nr=1
        r1 = -b/(2*a)
        r2 = r1

    end if
 
end subroutine quadratica

program segundograu
    implicit none
```

        ```fortran
        !         Resolve a equa√ß√£o d segundo grau:         !
        !                    ax^2+bx+c=0                    !
        !         Retornando apenas a ra√≠z positiva         !
        !---------------------------------------------------!
        implicit none

        real, intent(out) ::  tempo, dist
        real, parameter :: g = 9.8
        real :: delta,a,b,c
        a = g
        b = -vo*sin(angulo)
        c= -2*h

        if (abs(angulo) > 360.) then 
            angulo = mod(angulo,360.)

            tempo = (-b + sqrt(delta))/(2*a)
        else
            tempo = 0.
        end if
        
        dist = vo*cos(angulo)*tempo
        ```
